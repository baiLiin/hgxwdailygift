package com.cs37.gift;

import java.util.ArrayList;

public class gift {
    private String name;
    private int num;

    public gift() {

    }

    public gift(String name, int num) {
        this.num = num;
        this.name = name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getName() {
        return name;
    }

    public int getNum() {
        return num;
    }

    public void setNum(int num) {
        this.num = num;
    }
}
