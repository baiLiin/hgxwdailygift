package com.cs37.gift;

import java.util.ArrayList;
import java.util.Scanner;

public class getGift {

    public static void main(String[] args) {
        int chargeDay = 0;


        while (true) {

            System.out.println("请选择奖励类型: 0情缘1天天");
            Scanner sc = new Scanner(System.in);
            int choice = sc.nextInt();
            System.out.println("请输入玩家已充值天数");
            chargeDay = sc.nextInt();

            if (chargeDay < 0 || chargeDay >= 28) {
                System.out.println("充值天数大于希望天数,请检查!");
                continue;
            }

            if (choice == 0) {
                //获取情缘连冲补发奖励
                GetMethod.getLoveGift(chargeDay, GiftManager.loveGiftBox);
                break;
            } else if (choice == 1) {
                //获取天天连冲补发奖励
                GetMethod.getDailyGift(chargeDay, GiftManager.dailyGiftBox);
            } else {
                System.out.println("请选择正确的选项");
                continue;
            }
        }
    }
}
