package com.cs37.gift;

import java.util.ArrayList;


public class GetMethod {
    public static void getLoveGift(int chargeDay, ArrayList<gift> loveGiftBox) {
        gift band = loveGiftBox.get(0);
        gift stone = loveGiftBox.get(1);
        gift glass = loveGiftBox.get(2);
        gift special = loveGiftBox.get(3);
        if (chargeDay < 3) {
            //玩家充值3天,为他补发 3,7,15,21,28奖励
            System.out.println("为玩家补发:"
                    + band.getName() + "*" + band.getNum() * 4 + " | "
                    + stone.getName() + "*" + stone.getNum() * 14 + " | "
                    + glass.getName() + "*" + glass.getNum() * 14 + " | "
                    + special.getName() + "*" + band.getNum() * 1
            );
        } else if (chargeDay >= 3 && chargeDay < 7) {
            //玩家充值<=7天,为他补发7,,15,21,28奖励
            System.out.println("为玩家补发:"
                    + band.getName() + "*" + band.getNum() * 3 + " | "
                    + stone.getName() + "*" + stone.getNum() * 12 + " | "
                    + glass.getName() + "*" + glass.getNum() * 12 + " | "
                    + special.getName() + "*" + band.getNum() * 1
            );

        } else if (chargeDay >= 7 && chargeDay < 15) {
            //玩家充值<=15天,为他补发15,21,28奖励;
            System.out.println("为玩家补发:"
                    + band.getName() + "*" + band.getNum() * 2 + " | "
                    + stone.getName() + "*" + stone.getNum() * 10 + " | "
                    + glass.getName() + "*" + glass.getNum() * 10 + " | "
                    + special.getName() + "*" + band.getNum() * 1
            );

        } else if (chargeDay >= 15 && chargeDay < 21) {
            //玩家充值<=21天,为他补发21,28奖励
            System.out.println("为玩家补发:"
                    + band.getName() + "*" + band.getNum() * 1 + " | "
                    + stone.getName() + "*" + stone.getNum() * 7 + " | "
                    + glass.getName() + "*" + glass.getNum() * 7 + " | "
                    + special.getName() + "*" + band.getNum() * 1
            );
        } else if (chargeDay >= 21 && chargeDay < 28) {
            //玩家充值<= 28,为他补发28日奖励
            System.out.println("为玩家补发:"
                    + special.getName() + "*" + band.getNum() * 1 + " | "
                    + stone.getName() + "*" + stone.getNum() * 4 + " | "
                    + glass.getName() + "*" + glass.getNum() * 4
            );
        }
    }
